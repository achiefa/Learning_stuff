160 injector di_config::build_injector() {
161 injector inj;
162 std::stack<initializer_fn*> initializers; // ??
163
164 std::unordered_set<int> unmarked_nodes;
165 for ( auto &node : graph_ ) {
166 node.second.mark_ = node_info::mark::UNMARKED;
167 unmarked_nodes.insert(node.first);
168 }
169 while (!unmarked_nodes.empty()) {
170 int node_id = *unmarked_nodes.begin();
171 toposort_visit_node(node_id, &unmarked_nodes, &initializers);
172 }
173 while (!initializers.empty()) {
174 (*initializers.top())(inj);
175 initializers.pop();
176 }
177 return std::move(inj);

187 void di_config::toposort_visit_node(
188 int node_id,
189 std::unordered_set<int> *unmarked_nodes,
190 std::stack <di_config::initializer_fn*> *output) {
191 node_info &info = graph_[node_id];
192 if (info.mark_ == node_info::mark::TEMP) {
193 throw std::runtime_error(info.debug_type_name_ + " appears to be part of a cycle");
194 }
195 else if (info.mark_ == node_info::mark::UNMARKED) {
196 unmarked_nodes->erase(node_id);
197 info.mark_ = node_info::mark::TEMP;
198 for (int dependent : info.dependents_) {
199 toposort_visit_node(dependent, unmarked_nodes, output);
200 }
201 info.mark_ = node_info::mark::PERM;
202 if (info.has_initializer_) {
203 // if has_initializer_ is false, it means someone depends on this
204 // node, but an instance factory for this node has not been provided.
205 // This will result in an injection error later.
206 output->push(&info.initializer_);
207 }
208 }
